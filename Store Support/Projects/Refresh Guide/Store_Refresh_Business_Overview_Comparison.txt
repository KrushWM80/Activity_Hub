    // Render unified trend chart and week cards
    function renderTrendChart() {
        const container = document.getElementById("trendChart");
        
        COMPARISON_DATA.weeks.forEach((week, index) => {
            const prevWeek = index > 0 ? COMPARISON_DATA.weeks[index - 1] : null;
            const completion = parseFloat(week.summary.overallCompletionOfMax);
            const prevCompletion = prevWeek ? parseFloat(prevWeek.summary.overallCompletionOfMax) : 0;
            const change = prevWeek ? (completion - prevCompletion).toFixed(1) : completion.toFixed(1);
            const isWeek1 = index === 0;
            
            const column = document.createElement("div");
            column.className = "week-column";
            
            let changeHTML = "";
            if (isWeek1) {
                changeHTML = '<div class="week-change"><div class="change-value">+' + change + '%</div><div class="change-label">from 1/3/26</div></div>';
            } else {
                changeHTML = '<div class="week-change"><div class="change-value">+' + change + '%</div><div class="change-label">vs Previous Week</div></div>';
            }
            
            column.innerHTML = 
                '<div class="week-column-header">' + week.label + '</div>' +
                '<div class="week-column-date">' + week.date + '</div>' +
                '<div class="trend-percent">' + week.summary.overallCompletionOfMax + '%</div>' +
                '<div class="week-metrics">' +
                '<div class="metric-item"><div class="metric-item-value">' + (week.summary.totalCompletedItems / 1000).toFixed(0) + 'k</div><div class="metric-item-label">Completed</div></div>' +
                '<div class="metric-item"><div class="metric-item-value">' + (week.summary.storesWithAssignments / 1000).toFixed(1) + 'k</div><div class="metric-item-label">Stores Active</div></div>' +
                '</div>' +
                changeHTML +
                '</div>';
            
            container.appendChild(column);
        });
    }

    // Render user engagement metrics with percentages for ALL items and Total Actions
    function renderUserEngagement() {
        const container = document.getElementById("userEngagement");
        let html = "";
        
        COMPARISON_DATA.weeks.forEach((week, index) => {
            const prevWeek = index > 0 ? COMPARISON_DATA.weeks[index - 1].userEngagement : null;
            const current = week.userEngagement;
            
            // Calculate growth percentages for ALL metrics
            const workersGrowth = prevWeek && prevWeek.workers > 0 ? ((current.workers - prevWeek.workers) / prevWeek.workers * 100).toFixed(1) : null;
            const managersGrowth = prevWeek && prevWeek.managers > 0 ? ((current.managers - prevWeek.managers) / prevWeek.managers * 100).toFixed(1) : null;
            const totalUsersGrowth = prevWeek && prevWeek.totalUsers > 0 ? ((current.totalUsers - prevWeek.totalUsers) / prevWeek.totalUsers * 100).toFixed(1) : null;
            const assignmentsGrowth = prevWeek && prevWeek.assignments > 0 ? ((current.assignments - prevWeek.assignments) / prevWeek.assignments * 100).toFixed(1) : null;
            const completionsGrowth = prevWeek && prevWeek.completions > 0 ? ((current.completions - prevWeek.completions) / prevWeek.completions * 100).toFixed(1) : null;
            const totalActionsGrowth = prevWeek && prevWeek.totalActions > 0 ? ((current.totalActions - prevWeek.totalActions) / prevWeek.totalActions * 100).toFixed(1) : null;
            const actionsPerUserGrowth = prevWeek && prevWeek.actionsPerUser > 0 ? ((current.actionsPerUser - prevWeek.actionsPerUser) / prevWeek.actionsPerUser * 100).toFixed(1) : null;
            
            html += "<div class=\"engagement-card\">" +
                "<div class=\"engagement-header\">" + week.label + " (" + week.date + ")</div>" +
                
                // Total Actions - HIGHLIGHTED at top
                "<div class=\"engagement-metric highlight\">" +
                "<div class=\"metric-label\"><strong>📊 Total Actions</strong></div>" +
                "<div class=\"metric-value-wrapper\">" + 
                (totalActionsGrowth !== null ? "<div class=\"growth-badge\">+" + totalActionsGrowth + "%</div>" : "") + 
                "<div class=\"metric-value large\">" + (current.totalActions / 1000000).toFixed(2) + "M</div></div>" +
                "</div>" +
                
                // Unique Associates with %
                "<div class=\"engagement-metric\">" +
                "<div class=\"metric-label\">Unique Associates</div>" +
                "<div class=\"metric-value-wrapper\">" + 
                (workersGrowth !== null ? "<div class=\"growth-badge\">+" + workersGrowth + "%</div>" : "") + 
                "<div class=\"metric-value\">" + current.workers.toLocaleString() + "</div></div>" +
                "</div>" +
                
                // Unique Managers with %
                "<div class=\"engagement-metric\">" +
                "<div class=\"metric-label\">Unique Managers</div>" +
                "<div class=\"metric-value-wrapper\">" + 
                (managersGrowth !== null ? "<div class=\"growth-badge\">+" + managersGrowth + "%</div>" : "") + 
                "<div class=\"metric-value\">" + current.managers.toLocaleString() + "</div></div>" +
                "</div>" +
                
                // Total Users with %
                "<div class=\"engagement-metric\">" +
                "<div class=\"metric-label\">Total Users</div>" +
                "<div class=\"metric-value-wrapper\">" + 
                (totalUsersGrowth !== null ? "<div class=\"growth-badge\">+" + totalUsersGrowth + "%</div>" : "") + 
                "<div class=\"metric-value\">" + current.totalUsers.toLocaleString() + "</div></div>" +
                "</div>" +
                
                // Assignments with %
                "<div class=\"engagement-metric\">" +
                "<div class=\"metric-label\">Assignments</div>" +
                "<div class=\"metric-value-wrapper\">" + 
                (assignmentsGrowth !== null ? "<div class=\"growth-badge\">+" + assignmentsGrowth + "%</div>" : "") +

                "</div>" +
                
                // Completions with %
                "<div class=\"engagement-metric\">" +
                "<div class=\"metric-label\">Completions</div>" +
                "<div class=\"metric-value-wrapper\">" +

                "</div>" +
                
                "</div>";
        });
        
        container.innerHTML = html;
