The Date Dropdown gives users the ability to make a single date selection from a number of options.


<ScreenshotTabs screenshotData={{"default":"screenshots/date-dropdown.png","triggered":"screenshots/date-dropdown-triggered.png"}} baseUrl="/pages/electrode-mobile-platform/gtp-shared-components/"/>
## Usage
```js
import {DateDropdown} from '@walmart/gtp-shared-components`;

const [date, setDate] = React.useState<Date | undefined>(undefined);

<DateDropdown
  label={'Date'}
  value={date}
  helperText="Select one of these dates!"
  onSelect={(dt) => {
    if (dt) {
      setDate(dt);
    }
  }}
/>
```
