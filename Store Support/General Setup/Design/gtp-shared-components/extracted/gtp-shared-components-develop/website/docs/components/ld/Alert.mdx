## Usage
```js
import {Alert} from '@walmart/gtp-shared-components;

<Alert
  variant="info"
  children="This is an info alert"
  actionButtonProps={{
    children: 'Action',
    onPress: () => Alert.alert('Action', 'Action pressed'),
  }}
/>
```
