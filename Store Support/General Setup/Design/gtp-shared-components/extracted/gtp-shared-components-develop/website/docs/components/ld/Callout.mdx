## Usage
```js
import {Callout} from '@walmart/gtp-shared-components`;

const [isOpen, setIsOpen] = React.useState(false);

<Callout
  content={'Example Callout content.'}
  isOpen={isOpen}
  onClose={() => setIsOpen(false)}
  position="bottomRight">
  <Link
    onPress={() => setIsOpen(false)}>
    Bottom{'\n'}Right
  </Link>
</Callout>
```
